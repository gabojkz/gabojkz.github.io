#!/usr/bin/perl
my $prop = 'key';
my %hash = ( key => "some value");
print( $hash{$prop} );

#!/usr/bin/perl

my @names = ('juan', 'pedro', 'pablo');
my %address_book = (
  'juan'  => {
    city => 'manchester', country => 'UK'
  },
  'pedro' => {
    city => 'medellin', country => 'Colombia'
  },
  'pablo' => {
    city => 'santiago', country => 'Chile'
  },
);

foreach my $name (@names) {
  # by passing $name to $address_book is an easy way
  # to fetch the values
  my $address = $address_book{$name};
  print("$name lives in ");
  print("$address->{city}, $address->{country}");
  print "\n";
}

=pod
##result stdout 0s 4888KB
juan reside at manchester, UK
pedro reside at medellin, Colombia
pablo reside at santiago, Chile
=cut
